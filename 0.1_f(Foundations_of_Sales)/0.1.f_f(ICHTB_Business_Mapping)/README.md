def diagnose_organization(org: Organization) -> DiagnosisReport:
    """
    Assess health of each fan in the organization.
    """
    fans = {
        'delta_1_cio': assess_gradient(org.technology, org.innovation),
        'delta_2_chro': assess_curl(org.culture, org.retention),
        'delta_3_cfo': assess_expansion(org.capital, org.growth),
        'delta_4_coo': assess_compression(org.operations, org.delivery),
        'delta_5_cmo': assess_emergence(org.marketing, org.timing),
        'delta_6_ceo': assess_anchor(org.vision, org.alignment)
    }

    imbalances = []

    # Check complementary pairs
    if abs(fans['delta_1_cio'] - fans['delta_2_chro']) > 0.3:
        imbalances.append('CIO-CHRO imbalance: push without memory or vice versa')

    if abs(fans['delta_3_cfo'] - fans['delta_4_coo']) > 0.3:
        imbalances.append('CFO-COO imbalance: expansion without execution or vice versa')

    if abs(fans['delta_5_cmo'] - fans['delta_6_ceo']) > 0.3:
        imbalances.append('CMO-CEO imbalance: timing without purpose or vice versa')

    return DiagnosisReport(
        fan_scores=fans,
        imbalances=imbalances,
        healthy=len(imbalances) == 0,
        recommendation=generate_recommendation(fans, imbalances)
    )
